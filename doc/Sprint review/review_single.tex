{
	\vspace{0.5cm}
	\begin{center}
	\begin{Large}

		\hline
		\large{\textbf{Item}} & 
		\large{\textbf{Description}} & 
		\large{\textbf{Est. time}} &
		\large{\textbf{People}} \\
		\hline
	#1
	\end{tabular}

	\end{center}

	\setcounter{itemCounter}{0}
}
